## 3. Helper functions

clean_colnames <- function(df) {
  names(df) <- names(df) %>%
    str_replace_all("\\s+", "_") %>%
    str_replace_all("\\.", "_") %>%
    str_replace_all("-", "_")
  df
}

save_plot <- function(p, filename, width = 10, height = 7) {
  ggsave(
    filename = file.path(fig_dir, paste0(filename, ".pdf")),
    plot = p,
    width = width,
    height = height
  )
  
  ggsave(
    filename = file.path(fig_dir, paste0(filename, ".png")),
    plot = p,
    width = width,
    height = height,
    dpi = 300
  )
}

write_xlsx_safe <- function(x, filename) {
  openxlsx::write.xlsx(
    x,
    file = file.path(table_dir, filename),
    overwrite = TRUE
  )
}

get_row_value_by_set <- function(row_df, prefix, target_keyword) {
  cn <- names(row_df)
  hit <- cn[str_detect(cn, paste0("^", prefix, "_.*", target_keyword))]
  if (length(hit) == 0) {
    return(NA_real_)
  }
  as.numeric(row_df[[hit[1]]])
}

classify_ecological_function <- function(x) {
  x_low <- str_to_lower(x)
  
  case_when(
    str_detect(x_low, "carbon metabolism|carbon fixation|glycolysis|citrate cycle|pyruvate|propanoate|butanoate|glyoxylate|dicarboxylate|starch|sucrose|galactose|fructose|mannose") ~
      "Carbon metabolism",
    
    str_detect(x_low, "methane|methan") ~
      "Methane / C1 metabolism",
    
    str_detect(x_low, "nitrogen|nitrate|nitrite|denitrification|ammonia") ~
      "Nitrogen cycling",
    
    str_detect(x_low, "sulfur|sulfate|sulfide|thiosulfate") ~
      "Sulfur cycling",
    
    str_detect(x_low, "phosphorus|phosphonate|phosphate|phosphotransferase") ~
      "Phosphorus cycling",
    
    str_detect(x_low, "aromatic|benzoate|toluene|xylene|dioxin|naphthalene|polycyclic|chlorocyclohexane|chlorobenzene|aminobenzoate|caprolactam|ethylbenzene|styrene|nitrotoluene|bisphenol|xenobiotics") ~
      "Aromatic / xenobiotic degradation",
    
    str_detect(x_low, "secondary metabolite|polyketide|terpenoid|alkaloid|phenylpropanoid|flavonoid|ansamycin|streptomycin|novobiocin|vancomycin|betalain|zeatin") ~
      "Secondary metabolism",
    
    str_detect(x_low, "abc transporter|transporter|secretion|bacterial secretion|phosphotransferase system|pts") ~
      "Transport and secretion",
    
    str_detect(x_low, "two-component|quorum|biofilm|chemotaxis|flagellar|motility") ~
      "Signal transduction and motility",
    
    str_detect(x_low, "ribosome|dna replication|rna polymerase|homologous recombination|mismatch repair|base excision|nucleotide excision|aminoacyl|protein export|proteasome") ~
      "Core cellular processes",
    
    str_detect(x_low, "oxidative phosphorylation|photosynthesis|energy metabolism|electron transport|sulfur relay") ~
      "Energy metabolism",
    
    str_detect(x_low, "beta-lactam|drug resistance|antimicrobial|multidrug|efflux|cationic antimicrobial|resistance") ~
      "Stress resistance",
    
    str_detect(x_low, "amino acid|alanine|aspartate|glutamate|glycine|serine|threonine|cysteine|methionine|valine|leucine|isoleucine|lysine|arginine|proline|histidine|phenylalanine|tyrosine|tryptophan") ~
      "Amino acid metabolism",
    
    str_detect(x_low, "fatty acid|lipid|glycerolipid|glycerophospholipid|sphingolipid|linoleic|arachidonic") ~
      "Lipid metabolism",
    
    str_detect(x_low, "cofactor|vitamin|folate|biotin|riboflavin|porphyrin|thiamine|nicotinate|nicotinamide|pantothenate|coa") ~
      "Cofactor and vitamin metabolism",
    
    TRUE ~ "Other metabolism"
  )
}

raw_pathway <- readxl::read_excel(pathway_file) %>%
  as.data.frame()

pathway_df <- raw_pathway %>%
  clean_colnames()

required_cols <- c(
  "Pair",
  "Pathway_name",
  "Difference_in_KO_number",
  "Difference_in_proportion",
  "log2FC_proportion",
  "Higher_in"
)

missing_cols <- setdiff(required_cols, names(pathway_df))

if (length(missing_cols) > 0) {
  stop(
    "Overall_pairwise_KEGG_pathway_comparison_summary.xlsx is missing these columns: ",
    paste(missing_cols, collapse = ", ")
  )
}

pathway_core <- pathway_df %>%
  filter(Pair == target_pair)

if (nrow(pathway_core) == 0) {
  stop("Pair was not found in the pathway summary: ", target_pair)
}

## Source difference for Core_Native_vs_Invasive = Native - Invasive
## Multiply by -1 to standardize as Invasive - Native

pathway_clean <- pathway_core %>%
  rowwise() %>%
  mutate(
    Invasive_minus_Native_prop = -1 * Difference_in_proportion,
    Invasive_minus_Native_log2FC = -1 * log2FC_proportion,
    Invasive_minus_Native_KO_number = -1 * Difference_in_KO_number,
    
    Invasive_KO_number = get_row_value_by_set(cur_data(), "KO_number", "Invasive"),
    Native_KO_number = get_row_value_by_set(cur_data(), "KO_number", "Native"),
    Invasive_proportion = get_row_value_by_set(cur_data(), "Proportion", "Invasive"),
    Native_proportion = get_row_value_by_set(cur_data(), "Proportion", "Native"),
    
    Total_KO_number = sum(c(Invasive_KO_number, Native_KO_number), na.rm = TRUE),
    Mean_proportion = mean(c(Invasive_proportion, Native_proportion), na.rm = TRUE),
    Abs_difference = abs(Invasive_minus_Native_prop),
    
    Direction = case_when(
      Invasive_minus_Native_prop > 0 ~ "Invasive core KO higher",
      Invasive_minus_Native_prop < 0 ~ "Native core KO higher",
      TRUE ~ "Equal"
    ),
    
    Ecological_category = classify_ecological_function(Pathway_name)
  ) %>%
  ungroup()

write_xlsx_safe(
  pathway_clean,
  "00_Core_pathway_clean_Invasive_minus_Native.xlsx"
)

## Figure 06. Ecological functional category summary plot

category_long <- pathway_clean %>%
  select(
    Pathway_name,
    Ecological_category,
    Invasive_KO_number,
    Native_KO_number
  ) %>%
  pivot_longer(
    cols = c(Invasive_KO_number, Native_KO_number),
    names_to = "Group",
    values_to = "KO_number"
  ) %>%
  mutate(
    Group = case_when(
      Group == "Invasive_KO_number" ~ "Invasive core KO",
      Group == "Native_KO_number" ~ "Native core KO",
      TRUE ~ Group
    )
  ) %>%
  group_by(Group, Ecological_category) %>%
  summarise(
    KO_number = sum(KO_number, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  group_by(Group) %>%
  mutate(
    Total_KO_number = sum(KO_number, na.rm = TRUE),
    Proportion = KO_number / Total_KO_number
  ) %>%
  ungroup()

category_compare <- category_long %>%
  select(Group, Ecological_category, KO_number, Proportion) %>%
  pivot_wider(
    names_from = Group,
    values_from = c(KO_number, Proportion),

  mutate(
    Invasive_minus_Native_prop =
      `Proportion_Invasive core KO` - `Proportion_Native core KO`,
    Invasive_minus_Native_KO_number =
      `KO_number_Invasive core KO` - `KO_number_Native core KO`,
    Total_KO_number =
      `KO_number_Invasive core KO` + `KO_number_Native core KO`,

add_sheet <- function(wb, sheet_name, data) {
