#!/usr/bin/env Rscript

# ==============================================================================
# Cross-study alpha- and beta-diversity analysis for paired invasive/native
# rhizosphere microbiome datasets
#
# Designed for GitHub and reproducible use with either 16S or ITS OTU/ASV tables.
#
# Expected file names:
#   I16S-文章1.xlsx / N16S-文章1.xlsx
#   IITS-文章1.xlsx / NITS-文章1.xlsx
#
# Core statistical design
# ------------------------------------------------------------------------------
# 1. Invasive and native tables are merged within each study before analysis.
# 2. Alpha diversity is calculated after removing only all-zero features.
# 3. Beta diversity is calculated after joint within-study feature filtering
#    and total-sum scaling to relative abundance.
# 4. Samples are nested within PlantGroup; PlantGroup is nested within Study.
# 5. Alpha inference uses within-study z scores and mixed-effects models:
#
#      Alpha_z ~ Status + (1 | Study) + (1 | PlantGroup)
#
# 6. Study-level raw alpha differences and Hedges' g are calculated from
#    equally weighted PlantGroup means, not pooled samples.
# 7. Beta analyses use PlantGroup centroid compositions as ecological units.
# 8. Corrected Delta Bray-Curtis is:
#
#      mean(between-status Bray distance)
#      - mean(mean(within-invasive), mean(within-native))
#
# 9. Study-level PERMANOVA and beta dispersion are performed on PlantGroup
#    centroid Bray-Curtis distances.
# 10. Optional global partial dbRDA controls for Study:
#
#      community ~ Status + Condition(Study)
#
# Direction convention
# ------------------------------------------------------------------------------
# All contrasts are Invasive - Native.
# Positive alpha estimates indicate higher diversity under invasive plants.
# Positive Delta Bray-Curtis indicates that between-status dissimilarity
# exceeds average within-status heterogeneity.
#
# IMPORTANT
# ------------------------------------------------------------------------------
# PlantGroup is inferred from sample names unless a manual metadata file is
# supplied. Always inspect the exported metadata diagnostic before interpreting
# results.
# ==============================================================================

p_to_star <- function(p) {
  dplyr::case_when(
    is.na(p) ~ "NA",
    p < 0.001 ~ "***",
    p < 0.01 ~ "**",
    p < 0.05 ~ "*",
    TRUE ~ "ns"
  )
}

format_p <- function(p, adjusted = FALSE) {
  prefix <- if (adjusted) "BH-adjusted P " else "P "

  if (is.na(p)) {
    return(paste0(prefix, "= NA"))
  }

  if (p < 0.001) {
    return(paste0(prefix, "< 0.001"))
  }

  paste0(
    prefix,
    "= ",
    formatC(p, digits = 3, format = "f")
  )
}

safe_mean <- function(x) {
  x <- x[is.finite(x)]
  if (length(x) == 0) return(NA_real_)
  mean(x)
}

safe_sd <- function(x) {
  x <- x[is.finite(x)]
  if (length(x) < 2) return(NA_real_)
  stats::sd(x)
}

parse_input_file <- function(path, marker) {
  file_name <- tools::file_path_sans_ext(basename(path))
  article_number <- stringr::str_extract(
    file_name,
    "(?<=文章)\\d+"
  )

  tibble::tibble(
    FilePath = path,
    FileName = file_name,
    Study = ifelse(
      is.na(article_number),
      NA_character_,
      paste0("R", article_number)
    ),
    StudyNumber = suppressWarnings(as.numeric(article_number)),
    Status = dplyr::case_when(
      stringr::str_detect(
        file_name,
        paste0("^I", marker, "-文章\\d+$")
      ) ~ "Invasive",
      stringr::str_detect(
        file_name,
        paste0("^N", marker, "-文章\\d+$")
      ) ~ "Native",
      TRUE ~ NA_character_
    )
  )
}

infer_plant_group <- function(sample_id) {
  terminal_replicate_pattern <- stringr::regex(
    "([._-]?(rep|sample|s))?[._-]?\\d+$",
    ignore_case = TRUE
  )

  parsed <- stringr::str_replace(
    as.character(sample_id),
    terminal_replicate_pattern,
    ""
  )

  parsed <- stringr::str_replace(
    parsed,
    "[._-]+$",
    ""
  )

  invalid <- is.na(parsed) | parsed == ""
  parsed[invalid] <- as.character(sample_id)[invalid]

  parsed
}

read_manual_metadata <- function(path) {
  if (!file.exists(path)) {
    return(NULL)
  }

  sheets <- readxl::excel_sheets(path)

  if (!"metadata" %in% sheets) {
    stop(
      "Manual metadata file exists, but sheet 'metadata' is missing: ",
      path
    )
  }

  metadata <- readxl::read_excel(
    path,
    sheet = "metadata"
  ) |>
    as.data.frame()

  required_columns <- c(
    "Study",
    "SampleID",
    "Status",
    "PlantGroup"
  )

  missing_columns <- setdiff(
    required_columns,
    colnames(metadata)
  )

  if (length(missing_columns) > 0) {
    stop(
      "Manual metadata is missing columns: ",
      paste(missing_columns, collapse = ", ")
    )
  }

  metadata |>
    transmute(
      Study = as.character(Study),
      SampleID = as.character(SampleID),
      ManualStatus = as.character(Status),
      ManualPlantGroup = as.character(PlantGroup)
    ) |>
    distinct()
}

read_otu_table <- function(path) {
  dat <- readxl::read_excel(path) |>
    as.data.frame(check.names = FALSE)

  if (ncol(dat) < 2) {
    stop("Too few columns in ", basename(path))
  }

  colnames(dat)[1] <- "FeatureID"

  taxonomy_columns <- c(
    "taxonomy",
    "tax",
    "kingdom",
    "phylum",
    "class",
    "order",
    "family",
    "genus",
    "species"
  )

  candidate_columns <- setdiff(
    seq_along(dat),
    c(
      1,
      which(
        tolower(colnames(dat)) %in%
          taxonomy_columns
      )
    )
  )

  numeric_sample_columns <- vapply(
    dat[, candidate_columns, drop = FALSE],
    function(x) {
      converted <- suppressWarnings(as.numeric(x))
      mean(!is.na(converted)) >= 0.80
    },
    FUN.VALUE = logical(1)
  )

  sample_columns <- candidate_columns[
    numeric_sample_columns
  ]

  if (length(sample_columns) < 2) {
    stop(
      "Could not identify at least two numeric sample columns in ",
      basename(path)
    )
  }

  otu <- dat[, c(1, sample_columns), drop = FALSE]
  otu$FeatureID <- as.character(otu$FeatureID)

  for (j in 2:ncol(otu)) {
    otu[[j]] <- suppressWarnings(as.numeric(otu[[j]]))
  }

  otu[is.na(otu)] <- 0

  sample_names <- colnames(otu)[-1]

  otu |>
    group_by(FeatureID) |>
    summarise(
      across(
        all_of(sample_names),
        ~ sum(.x, na.rm = TRUE)
      ),
      .groups = "drop"
    )
}

merge_study_tables <- function(
  invasive_file,
  native_file,
  study
) {
  invasive_otu <- read_otu_table(invasive_file)
  native_otu <- read_otu_table(native_file)

  invasive_samples <- colnames(invasive_otu)[-1]
  native_samples <- colnames(native_otu)[-1]

  duplicated_samples <- intersect(
    invasive_samples,
    native_samples
  )

  if (length(duplicated_samples) > 0) {
    stop(
      study,
      ": sample names occur in both invasive and native tables: ",
      paste(duplicated_samples, collapse = ", ")
    )
  }

  merged <- dplyr::full_join(
    invasive_otu,
    native_otu,
    by = "FeatureID"
  )

  merged[is.na(merged)] <- 0

  count_matrix <- as.matrix(
    merged[, -1, drop = FALSE]
  )

  storage.mode(count_matrix) <- "numeric"
  rownames(count_matrix) <- merged$FeatureID

  count_matrix <- count_matrix[
    rowSums(count_matrix) > 0,
    ,
    drop = FALSE
  ]

  metadata <- tibble::tibble(
    Study = study,
    SampleID = c(
      invasive_samples,
      native_samples
    ),
    Status = c(
      rep("Invasive", length(invasive_samples)),
      rep("Native", length(native_samples))
    ),
    AutoPlantGroup = infer_plant_group(
      c(invasive_samples, native_samples)
    )
  )

  if (!is.null(manual_metadata)) {
    metadata <- metadata |>
      left_join(
        manual_metadata,
        by = c("Study", "SampleID")
      ) |>
      mutate(
        PlantGroup_raw = ifelse(
          !is.na(ManualPlantGroup) &
            ManualPlantGroup != "",
          ManualPlantGroup,
          AutoPlantGroup
        )
      )

    status_mismatch <- metadata |>
      filter(
        !is.na(ManualStatus),
        ManualStatus != Status
      )

    if (nrow(status_mismatch) > 0) {
      stop(
        study,
        ": manual Status disagrees with file prefix."
      )
    }
  } else {
    metadata <- metadata |>
      mutate(
        PlantGroup_raw = AutoPlantGroup
      )
  }

  metadata <- metadata |>
    mutate(
      Status = factor(
        Status,
        levels = c("Native", "Invasive")
      ),
      PlantGroup = paste(
        Study,
        PlantGroup_raw,
        sep = "__"
      )
    )

  sample_match <- match(
    colnames(count_matrix),
    metadata$SampleID
  )

  if (anyNA(sample_match)) {
    stop(
      study,
      ": failed to match matrix samples to metadata."
    )
  }

  metadata <- metadata[
    sample_match,
    ,
    drop = FALSE
  ]

  list(
    count_matrix = count_matrix,
    metadata = metadata
  )
}

# ==============================================================================

prepare_alpha_matrix <- function(
  count_matrix,
  metadata
) {
  library_size <- colSums(count_matrix)

  depth_table <- metadata |>
    mutate(
      LibrarySize = as.numeric(
        library_size[SampleID]
      ),
      RetainedForAlpha =
        LibrarySize >= MIN_LIBRARY_SIZE
    )

  alpha_matrix <- count_matrix
  rarefaction_depth <- NA_real_

  if (RAREFY_ALPHA) {
    retained_samples <- depth_table |>
      filter(RetainedForAlpha) |>
      pull(SampleID)

    alpha_matrix <- alpha_matrix[
      ,
      retained_samples,
      drop = FALSE
    ]

    rarefaction_depth <- min(
      colSums(alpha_matrix)
    )

    alpha_matrix <- t(
      vegan::rrarefy(
        t(alpha_matrix),
        sample = rarefaction_depth
      )
    )
  }

  list(
    alpha_matrix = alpha_matrix,
    depth_table = depth_table,
    rarefaction_depth = rarefaction_depth
  )
}

calculate_alpha <- function(
  alpha_matrix,
  metadata
) {
  community <- t(alpha_matrix)

  tibble::tibble(
    SampleID = rownames(community),
    Richness = as.numeric(
      vegan::specnumber(community)
    ),
    Chao1 = as.numeric(
      vegan::estimateR(community)["S.chao1", ]
    ),
    Shannon = as.numeric(
      vegan::diversity(
        community,
        index = "shannon"
      )
    ),
    Simpson = as.numeric(
      vegan::diversity(
        community,
        index = "simpson"
      )
    )
  ) |>
    left_join(
      metadata,
      by = "SampleID"
    )
}

calculate_hedges_g <- function(
  invasive_values,
  native_values
) {
  invasive_values <- invasive_values[
    is.finite(invasive_values)
  ]

  native_values <- native_values[
    is.finite(native_values)
  ]

  n_invasive <- length(invasive_values)
  n_native <- length(native_values)

  if (
    n_invasive < 2 ||
      n_native < 2
  ) {
    return(
      tibble::tibble(
        n_Invasive_PlantGroups = n_invasive,
        n_Native_PlantGroups = n_native,
        mean_Invasive = safe_mean(invasive_values),
        mean_Native = safe_mean(native_values),
        Delta = NA_real_,
        Hedges_g = NA_real_,
        var_g = NA_real_,
        ci_low = NA_real_,
        ci_high = NA_real_
      )
    )
  }

  mean_invasive <- mean(invasive_values)
  mean_native <- mean(native_values)
  sd_invasive <- stats::sd(invasive_values)
  sd_native <- stats::sd(native_values)

  pooled_sd <- sqrt(
    (
      (n_invasive - 1) * sd_invasive^2 +
        (n_native - 1) * sd_native^2
    ) /
      (
        n_invasive +
          n_native -
          2
      )
  )

  if (
    !is.finite(pooled_sd) ||
      pooled_sd <= 0
  ) {
    return(
      tibble::tibble(
        n_Invasive_PlantGroups = n_invasive,
        n_Native_PlantGroups = n_native,
        mean_Invasive = mean_invasive,
        mean_Native = mean_native,
        Delta = mean_invasive - mean_native,
        Hedges_g = NA_real_,
        var_g = NA_real_,
        ci_low = NA_real_,
        ci_high = NA_real_
      )
    )
  }

  cohen_d <- (
    mean_invasive -
      mean_native
  ) / pooled_sd

  correction_j <- 1 -
    3 /
      (
        4 *
          (
            n_invasive +
              n_native
          ) -
          9
      )

  hedges_g <- correction_j * cohen_d

  variance_g <- (
    n_invasive +
      n_native
  ) /
    (
      n_invasive *
        n_native
    ) +
    hedges_g^2 /
      (
        2 *
          (
            n_invasive +
              n_native -
              2
          )
      )

  standard_error <- sqrt(variance_g)

  tibble::tibble(
    n_Invasive_PlantGroups = n_invasive,
    n_Native_PlantGroups = n_native,
    mean_Invasive = mean_invasive,
    mean_Native = mean_native,
    Delta = mean_invasive - mean_native,
    Hedges_g = hedges_g,
    var_g = variance_g,
    ci_low = hedges_g - 1.96 * standard_error,
    ci_high = hedges_g + 1.96 * standard_error
  )
}

# 8. BETA-DIVERSITY FUNCTIONS
# ==============================================================================

prepare_beta_relative_abundance <- function(
  count_matrix
) {
  keep_feature <- (
    rowSums(count_matrix) >=
      BETA_MIN_TOTAL_COUNT &
      rowSums(count_matrix > 0) >=
        BETA_MIN_PREVALENCE
  )

  filtered_matrix <- count_matrix[
    keep_feature,
    ,
    drop = FALSE
  ]

  if (nrow(filtered_matrix) == 0) {
    stop(
      "No features remained after beta-diversity filtering."
    )
  }

  sample_by_feature <- t(filtered_matrix)

  if (any(rowSums(sample_by_feature) == 0)) {
    stop(
      "At least one sample has zero abundance after filtering."
    )
  }

  vegan::decostand(
    sample_by_feature,
    method = "total"
  )
}

calculate_plantgroup_centroids <- function(
  relative_abundance,
  metadata
) {
  community_data <- as.data.frame(
    relative_abundance,
    check.names = FALSE
  ) |>
    tibble::rownames_to_column(
      "SampleID"
    ) |>
    left_join(
      metadata,
      by = "SampleID"
    )

  feature_columns <- setdiff(
    colnames(community_data),
    c(
      "SampleID",
      "Study",
      "Status",
      "AutoPlantGroup",
      "PlantGroup_raw",
      "PlantGroup",
      "ManualStatus",
      "ManualPlantGroup"
    )
  )

  centroid_data <- community_data |>
    group_by(
      Study,
      Status,
      PlantGroup,
      PlantGroup_raw
    ) |>
    summarise(
      across(
        all_of(feature_columns),
        ~ mean(.x, na.rm = TRUE)
      ),
      n_samples = n(),
      .groups = "drop"
    )

  centroid_matrix <- centroid_data |>
    select(
      all_of(feature_columns)
    ) |>
    as.matrix()

  storage.mode(centroid_matrix) <- "numeric"
  rownames(centroid_matrix) <- centroid_data$PlantGroup

  list(
    centroid_matrix = centroid_matrix,
    centroid_metadata = centroid_data |>
      select(
        Study,
        Status,
        PlantGroup,
        PlantGroup_raw,
        n_samples
      ),
    centroid_data = centroid_data
  )
}

calculate_delta_bray <- function(
  centroid_matrix,
  status_vector
) {
  status_vector <- as.character(status_vector)

  invasive_index <- which(
    status_vector == "Invasive"
  )

  native_index <- which(
    status_vector == "Native"
  )

  if (
    length(invasive_index) < 2 ||
      length(native_index) < 2
  ) {
    return(
      tibble::tibble(
        MeanBetween = NA_real_,
        MeanWithinInvasive = NA_real_,
        MeanWithinNative = NA_real_,
        MeanWithinAverage = NA_real_,
        Delta_BrayCurtis = NA_real_
      )
    )
  }

  bray_matrix <- as.matrix(
    vegan::vegdist(
      centroid_matrix,
      method = "bray"
    )
  )

  invasive_within <- bray_matrix[
    invasive_index,
    invasive_index,
    drop = FALSE
  ]

  native_within <- bray_matrix[
    native_index,
    native_index,
    drop = FALSE
  ]

  between <- bray_matrix[
    invasive_index,
    native_index,
    drop = FALSE
  ]

  mean_within_invasive <- mean(
    invasive_within[
      upper.tri(invasive_within)
    ],
    na.rm = TRUE
  )

  mean_within_native <- mean(
    native_within[
      upper.tri(native_within)
    ],
    na.rm = TRUE
  )

  mean_between <- mean(
    between,
    na.rm = TRUE
  )

  mean_within_average <- mean(
    c(
      mean_within_invasive,
      mean_within_native
    )
  )

  tibble::tibble(
    MeanBetween = mean_between,
    MeanWithinInvasive = mean_within_invasive,
    MeanWithinNative = mean_within_native,
    MeanWithinAverage = mean_within_average,
    Delta_BrayCurtis =
      mean_between -
      mean_within_average
  )
}

bootstrap_delta_bray <- function(
  centroid_matrix,
  status_vector,
  n_boot = 999
) {
  status_vector <- as.character(status_vector)

  invasive_index <- which(
    status_vector == "Invasive"
  )

  native_index <- which(
    status_vector == "Native"
  )

  if (
    length(invasive_index) < 2 ||
      length(native_index) < 2
  ) {
    return(
      list(
        values = numeric(0),
        SE = NA_real_,
        CI_low = NA_real_,
        CI_high = NA_real_
      )
    )
  }

  bootstrap_values <- replicate(
    n_boot,
    {
      sampled_invasive <- sample(
        invasive_index,
        size = length(invasive_index),
        replace = TRUE
      )

      sampled_native <- sample(
        native_index,
        size = length(native_index),
        replace = TRUE
      )

      bootstrap_matrix <- rbind(
        centroid_matrix[
          sampled_invasive,
          ,
          drop = FALSE
        ],
        centroid_matrix[
          sampled_native,
          ,
          drop = FALSE
        ]
      )

      bootstrap_status <- c(
        rep("Invasive", length(sampled_invasive)),
        rep("Native", length(sampled_native))
      )

      calculate_delta_bray(
        bootstrap_matrix,
        bootstrap_status
      )$Delta_BrayCurtis
    }
  )

  bootstrap_values <- bootstrap_values[
    is.finite(bootstrap_values)
  ]

  if (length(bootstrap_values) < 20) {
    return(
      list(
        values = bootstrap_values,
        SE = NA_real_,
        CI_low = NA_real_,
        CI_high = NA_real_
      )
    )
  }

  list(
    values = bootstrap_values,
    SE = stats::sd(bootstrap_values),
    CI_low = unname(
      stats::quantile(
        bootstrap_values,
        probs = 0.025,
        na.rm = TRUE,
        type = 7
      )
    ),
    CI_high = unname(
      stats::quantile(
        bootstrap_values,
        probs = 0.975,
        na.rm = TRUE,
        type = 7
      )
    )
  )
}

permutation_delta_bray_p <- function(
  centroid_matrix,
  status_vector,
  observed_delta,
  n_perm = 999
) {
  if (!is.finite(observed_delta)) {
    return(NA_real_)
  }

  status_vector <- as.character(status_vector)

  permutation_values <- replicate(
    n_perm,
    {
      permuted_status <- sample(
        status_vector,
        replace = FALSE
      )

      calculate_delta_bray(
        centroid_matrix,
        permuted_status
      )$Delta_BrayCurtis
    }
  )

  permutation_values <- permutation_values[
    is.finite(permutation_values)
  ]

  if (length(permutation_values) == 0) {
    return(NA_real_)
  }

  (
    sum(
      abs(permutation_values) >=
        abs(observed_delta)
    ) + 1
  ) /
    (
      length(permutation_values) + 1
    )
}

calculate_permanova <- function(
  centroid_matrix,
  centroid_metadata
) {
  n_invasive <- sum(
    centroid_metadata$Status == "Invasive"
  )

  n_native <- sum(
    centroid_metadata$Status == "Native"
  )

  if (
    n_invasive < 2 ||
      n_native < 2
  ) {
    return(
      tibble::tibble(
        PERMANOVA_F = NA_real_,
        PERMANOVA_R2 = NA_real_,
        PERMANOVA_P = NA_real_
      )
    )
  }

  bray_distance <- vegan::vegdist(
    centroid_matrix,
    method = "bray"
  )

  result <- try(
    vegan::adonis2(
      bray_distance ~ Status,
      data = centroid_metadata,
      permutations = N_PERM
    ),
    silent = TRUE
  )

  if (inherits(result, "try-error")) {
    return(
      tibble::tibble(
        PERMANOVA_F = NA_real_,
        PERMANOVA_R2 = NA_real_,
        PERMANOVA_P = NA_real_
      )
    )
  }

  p_column <- grep(
    "^Pr",
    colnames(result),
    value = TRUE
  )[1]

  tibble::tibble(
    PERMANOVA_F = as.numeric(
      result["Status", "F"]
    ),
    PERMANOVA_R2 = as.numeric(
      result["Status", "R2"]
    ),
    PERMANOVA_P = as.numeric(
      result["Status", p_column]
    )
  )
}

calculate_beta_dispersion <- function(
  centroid_matrix,
  centroid_metadata
) {
  n_invasive <- sum(
    centroid_metadata$Status == "Invasive"
  )

  n_native <- sum(
    centroid_metadata$Status == "Native"
  )

  if (
    n_invasive < 2 ||
      n_native < 2
  ) {
    return(
      list(
        values = tibble::tibble(),
        summary = tibble::tibble(
          Dispersion_Invasive = NA_real_,
          Dispersion_Native = NA_real_,
          Delta_Dispersion = NA_real_,
          PERMDISP_F = NA_real_,
          PERMDISP_P = NA_real_
        )
      )
    )
  }

  bray_distance <- vegan::vegdist(
    centroid_matrix,
    method = "bray"
  )

  dispersion <- vegan::betadisper(
    bray_distance,
    group = factor(
      centroid_metadata$Status,
      levels = c("Native", "Invasive")
    ),
    type = "centroid",
    bias.adjust = TRUE
  )

  dispersion_values <- tibble::tibble(
    PlantGroup = names(dispersion$distances),
    DistanceToStatusCentroid = as.numeric(
      dispersion$distances
    )
  ) |>
    left_join(
      centroid_metadata,
      by = "PlantGroup"
    )

  dispersion_invasive <- mean(
    dispersion_values$DistanceToStatusCentroid[
      dispersion_values$Status == "Invasive"
    ],
    na.rm = TRUE
  )

  dispersion_native <- mean(
    dispersion_values$DistanceToStatusCentroid[
      dispersion_values$Status == "Native"
    ],
    na.rm = TRUE
  )

  permdisp <- try(
    vegan::permutest(
      dispersion,
      permutations = N_PERM
    ),
    silent = TRUE
  )

  permdisp_f <- NA_real_
  permdisp_p <- NA_real_

  if (!inherits(permdisp, "try-error")) {
    permdisp_table <- as.data.frame(permdisp$tab)
    p_column <- grep(
      "^Pr",
      colnames(permdisp_table),
      value = TRUE
    )[1]

    permdisp_f <- as.numeric(
      permdisp_table[1, "F"]
    )

    permdisp_p <- as.numeric(
      permdisp_table[1, p_column]
    )
  }

  list(
    values = dispersion_values,
    summary = tibble::tibble(
      Dispersion_Invasive = dispersion_invasive,
      Dispersion_Native = dispersion_native,
      Delta_Dispersion =
        dispersion_invasive -
        dispersion_native,
      PERMDISP_F = permdisp_f,
      PERMDISP_P = permdisp_p
    )
  )
}

calculate_pcoa <- function(
  centroid_matrix,
  centroid_metadata,
  study
) {
  if (nrow(centroid_matrix) < 3) {
    return(tibble::tibble())
  }

  bray_distance <- vegan::vegdist(
    centroid_matrix,
    method = "bray"
  )

  pcoa <- try(
    stats::cmdscale(
      bray_distance,
      k = min(
        2,
        nrow(centroid_matrix) - 1
      ),
      eig = TRUE,
      add = TRUE
    ),
    silent = TRUE
  )

  if (inherits(pcoa, "try-error")) {
    return(tibble::tibble())
  }

  coordinates <- as.matrix(pcoa$points)

  if (ncol(coordinates) == 1) {
    coordinates <- cbind(
      coordinates,
      PCoA2 = 0
    )
  }

  positive_eigenvalues <- pcoa$eig[
    pcoa$eig > 0
  ]

  axis_1_percent <- ifelse(
    length(positive_eigenvalues) >= 1,
    positive_eigenvalues[1] /
      sum(positive_eigenvalues) * 100,
    NA_real_
  )

  axis_2_percent <- ifelse(
    length(positive_eigenvalues) >= 2,
    positive_eigenvalues[2] /
      sum(positive_eigenvalues) * 100,
    NA_real_
  )

  tibble::tibble(
    PlantGroup = rownames(coordinates),
    PCoA1 = coordinates[, 1],
    PCoA2 = coordinates[, 2],
    PCoA1_percent = axis_1_percent,
    PCoA2_percent = axis_2_percent,
    Study = study
  ) |>
    left_join(
      centroid_metadata,
      by = c("Study", "PlantGroup")
    )
}

fit_alpha_model <- function(metric_name) {
  metric_data <- alpha_sample_z |>
    filter(
      Metric == metric_name
    ) |>
    mutate(
      Status = factor(
        Status,
        levels = c("Native", "Invasive")
      ),
      Study = factor(Study),
      PlantGroup = factor(PlantGroup)
    )

  full_model <- try(
    lmerTest::lmer(
      Alpha_z ~
        Status +
        (1 | Study) +
        (1 | PlantGroup),
      data = metric_data,
      REML = TRUE
    ),
    silent = TRUE
  )

  if (!inherits(full_model, "try-error")) {
    model <- full_model
    model_formula <- paste0(
      "Alpha_z ~ Status + ",
      "(1|Study) + (1|PlantGroup)"
    )
  } else {
    model <- lmerTest::lmer(
      Alpha_z ~
        Status +
        (1 | Study),
      data = metric_data,
      REML = TRUE
    )

    model_formula <- paste0(
      "Alpha_z ~ Status + ",
      "(1|Study)"
    )
  }

  coefficients <- summary(model)$coefficients

  tibble::tibble(
    Metric = metric_name,
    Model = model_formula,
    Singular = lme4::isSingular(
      model,
      tol = 1e-5
    ),
    Estimate_Invasive_minus_Native =
      coefficients[
        "StatusInvasive",
        "Estimate"
      ],
    SE =
      coefficients[
        "StatusInvasive",
        "Std. Error"
      ],
    df =
      coefficients[
        "StatusInvasive",
        "df"
      ],
    t_value =
      coefficients[
        "StatusInvasive",
        "t value"
      ],
    P =
      coefficients[
        "StatusInvasive",
        "Pr(>|t|)"
      ]
  )
}

make_alpha_forest <- function(metric_name) {
  dat <- alpha_effects_all |>
    filter(
      Metric == metric_name,
      is.finite(Hedges_g),
      is.finite(var_g),
      var_g > 0,
      is.finite(ci_low),
      is.finite(ci_high)
    ) |>
    mutate(
      StudyNumber = study_number(Study)
    ) |>
    arrange(StudyNumber)

  if (nrow(dat) == 0) {
    return(
      ggplot() +
        annotate(
          "text",
          x = 0,
          y = 0,
          label = paste0(
            metric_name,
            ": no valid effects"
          )
        ) +
        theme_void()
    )
  }

  dat <- dat |>
    mutate(
      y = rev(seq_len(n()))
    )

  meta_fit <- NULL

  if (nrow(dat) >= 2) {
    meta_fit <- metafor::rma.uni(
      yi = dat$Hedges_g,
      vi = dat$var_g,
      method = "REML"
    )
  }

  plot_object <- ggplot(dat) +
    geom_vline(
      xintercept = 0,
      linetype = "dashed",
      color = "red",
      linewidth = 0.55
    ) +
    geom_segment(
      aes(
        x = ci_low,
        xend = ci_high,
        y = y,
        yend = y
      ),
      linewidth = 0.55
    ) +
    geom_point(
      aes(
        x = Hedges_g,
        y = y
      ),
      size = 2.3
    )

  y_breaks <- dat$y
  y_labels <- dat$Study
  subtitle_text <- paste0(
    nrow(dat),
    " studies"
  )

  if (!is.null(meta_fit)) {
    pooled <- tibble::tibble(
      Estimate = as.numeric(meta_fit$b),
      CI_low = meta_fit$ci.lb,
      CI_high = meta_fit$ci.ub,
      y = 0
    )

    plot_object <- plot_object +
      geom_segment(
        data = pooled,
        aes(
          x = CI_low,
          xend = CI_high,
          y = y,
          yend = y
        ),
        inherit.aes = FALSE,
        linewidth = 1.0
      ) +
      geom_point(
        data = pooled,
        aes(
          x = Estimate,
          y = y
        ),
        inherit.aes = FALSE,
        shape = 23,
        size = 4.0,
        fill = "white",
        stroke = 0.9
      )

    y_breaks <- c(0, y_breaks)
    y_labels <- c("Pooled", y_labels)

    subtitle_text <- paste0(
      "REML pooled g = ",
      formatC(
        as.numeric(meta_fit$b),
        digits = 2,
        format = "f"
      ),
      " [",
      formatC(
        meta_fit$ci.lb,
        digits = 2,
        format = "f"
      ),
      ", ",
      formatC(
        meta_fit$ci.ub,
        digits = 2,
        format = "f"
      ),
      "]; ",
      format_p(meta_fit$pval),
      "; I2 = ",
      formatC(
        meta_fit$I2,
        digits = 1,
        format = "f"
      ),
      "%"
    )
  }

  plot_object +
    scale_y_continuous(
      breaks = y_breaks,
      labels = y_labels,
      expand = expansion(
        mult = c(0.03, 0.04)
      )
    ) +
    labs(
      title = metric_labels[[metric_name]],
      subtitle = subtitle_text,
      x = "Hedges' g (Invasive - Native)",
      y = NULL
    ) +
    theme_publication +
    theme(
      panel.grid.major.y = element_blank()
    )
}

save_plot_set <- function(
  plot_object,
  file_stem,
  width,
  height
) {
  ggplot2::ggsave(
    filename = file.path(
      FIGURE_DIR,
      paste0(file_stem, ".pdf")
    ),
    plot = plot_object,
    width = width,
    height = height,
    units = "in"
  )

  ggplot2::ggsave(
    filename = file.path(
      FIGURE_DIR,
      paste0(file_stem, ".tiff")
    ),
    plot = plot_object,
    width = width,
    height = height,
    units = "in",
    dpi = 600,
    compression = "lzw"
  )

  ggplot2::ggsave(
    filename = file.path(
      FIGURE_DIR,
      paste0(file_stem, ".png")
    ),
    plot = plot_object,
    width = width,
    height = height,
    units = "in",
    dpi = 400
  )
}
